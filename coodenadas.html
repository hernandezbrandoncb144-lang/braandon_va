import cv2
#importa libreria opencv

#carga imagen
gato=cv2.imread('gatotito.jpg')

while True:

    recorte=gato[0:100,55:200]  #alto y ancho
    #mostrar
    cv2.imshow('imagen original',gato)
    cv2.imshow('imagen recortada',recorte)
    tecla = cv2.waitKey(0)
    if tecla==ord('A'):
        cv2.imshow('imagen original', gato)

    elif tecla == ord('Y'):
        cv2.imshow('imagen recortada', recorte)

    elif tecla==ord('X'):
        cv2.destroyAllWindows()
        break
